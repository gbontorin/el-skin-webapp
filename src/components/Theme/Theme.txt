//import React from "react";
import { createContext } from 'node:vm';
import { ThemeProvider } from 'styled-components';

const theme_light = {
  color: '#333333',
  backgroundColor: '#FFFFFF',
};

const theme_dark = {
  color:'#FFFFFF',
  backgroundColor: '#333333',
};

// Create a context to provide theme switching functionality
const ThemeToggleContext = createContext(null);

const Theme = ({ children }) => {
  const [currentTheme, setCurrentTheme] = useState(theme_light); // Default to light theme

  const toggleTheme = () => {
    setCurrentTheme((prevTheme) =>
      prevTheme === theme_light ? theme_dark : theme_light
    );
  };

  return (
    <ThemeToggleContext.Provider value={toggleTheme}>
      <ThemeProvider theme={currentTheme}>{children}</ThemeProvider>
    </ThemeToggleContext.Provider>
  );
};

// Custom hook to easily use the theme toggler in other components
export const useThemeToggle = () => {
  return useContext(ThemeToggleContext);
};

export default Theme;
